Replace the entire block above with:

```typescript
} : isValidJSON(message.content) ? (
  (() => {
    try {
      const output: StructuredOutput = JSON.parse(message.content);
      return (
        <StructuredOutputRenderer
          output={output}
          messageId={message.id}
          onSave={() => saveOutput(message.id, message.content)}
          onAdjust={(channelIndex) =>
            openAdjust({
              messageId: message.id,
              sectionTitle: output.channels[channelIndex]?.type ?? "channel",
              sectionBody: JSON.stringify(output.channels[channelIndex] ?? {}),
              outputType: "generic",
            })
          }
        />
      );
    } catch {
      return (
        <div className="text-sm whitespace-pre-wrap leading-relaxed">
          {message.content}
        </div>
      );
    }
  })()
)
```
