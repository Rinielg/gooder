    **onSubmit handler** — CRITICAL: read from `values.email`, `values.password`, `values.workspaceName`, NOT deleted state vars. Preserve the fallback workspace name logic:
    ```typescript
    async function onSubmit(values: RegisterFormValues) {
      setLoading(true)
      try {
        const { data: authData, error: authError } = await supabase.auth.signUp({
          email: values.email,
          password: values.password,
        })
        if (authError) { toast.error(authError.message); return }
        if (!authData.user) { toast.error("Failed to create account"); return }

        const res = await fetch("/api/auth/register", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({
            userId: authData.user.id,
            workspaceName: values.workspaceName || `${values.email.split("@")[0]}'s Workspace`,
          }),
        })
        if (!res.ok) {
          const errData = await res.json()
          toast.error(errData.error || "Failed to set up workspace")
          return
        }
        toast.success("Account created! Redirecting...")
        router.push("/chat")
        router.refresh()
      } catch {
        toast.error("An unexpected error occurred")
      } finally {
        setLoading(false)
      }
    }
    ```
