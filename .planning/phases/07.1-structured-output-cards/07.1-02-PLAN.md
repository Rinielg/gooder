**Imports:**
```typescript
import { useState, useRef, useEffect } from "react";
import { X } from "lucide-react";
import { Button } from "@/components/ui/button";
import { Badge } from "@/components/ui/badge";
import { cn } from "@/lib/utils";
import { AdjustTarget, OUTPUT_TYPE_CONFIG, DetectedOutputType } from "./output-card";
```

**Props interface:**
```typescript
interface AdjustDialogProps {
  target: AdjustTarget;
  onSubmit: (text: string) => void;
  onClose: () => void;
}
```

**Component:** `export function AdjustDialog({ target, onSubmit, onClose }: AdjustDialogProps)`

- Internal state: `const [text, setText] = useState("")` — per-keystroke value STAYS INSIDE this component. Do not expose as a controlled prop. This is the critical decision: `adjustText` must NOT live in page.tsx where it would re-render the message list on every keystroke.
- Autofocus ref: `const inputRef = useRef<HTMLTextAreaElement>(null)` with `useEffect(() => { inputRef.current?.focus() }, [])` — empty deps so it focuses only on mount.
- Config: `const config = OUTPUT_TYPE_CONFIG[target.outputType]`

**Layout — container** (matches input form width and elevation from page.tsx):
```typescript
<div className="w-[50vw] bg-background border border-border rounded-2xl shadow-elevation-3 px-4 py-3 space-y-3">
```

**Header row** — shows which output is being adjusted + close button:
```typescript
<div className="flex items-center justify-between">
  <div className="flex items-center gap-2 min-w-0">
    <span className="text-xs font-medium text-muted-foreground flex-shrink-0">Adjusting:</span>
    <Badge className={cn("text-xs border flex-shrink-0", config.badgeClass)}>
      {config.label}
    </Badge>
    <span className="text-xs text-muted-foreground truncate">{target.sectionTitle}</span>
  </div>
  <button
    onClick={onClose}
    className="flex-shrink-0 text-muted-foreground hover:text-foreground transition-colors ml-2"
    aria-label="Close adjust panel"
  >
    <X className="w-3.5 h-3.5" />
  </button>
</div>
```

**Input row** — textarea + Send button side by side:
```typescript
<div className="flex items-end gap-2">
  <textarea
    ref={inputRef}
    value={text}
    onChange={e => setText(e.target.value)}
    onKeyDown={e => {
      if (e.key === "Enter" && !e.shiftKey) {
        e.preventDefault();
        if (text.trim()) {
          onSubmit(text.trim());
          setText("");
        }
      }
    }}
    placeholder="What changes do you want? (Enter to send, Shift+Enter for newline)"
    rows={2}
    className="flex-1 resize-none bg-transparent text-sm focus:outline-none border border-border rounded-lg px-3 py-2 leading-relaxed"
  />
  <Button
    size="sm"
    onClick={() => {
      if (text.trim()) {
        onSubmit(text.trim());
        setText("");
      }
    }}
    disabled={!text.trim()}
    className="h-9 flex-shrink-0"
  >
    Send
  </Button>
</div>
```

Note: `onSubmit` receives the trimmed text and the component resets `text` to `""`. The parent (page.tsx) never holds `adjustText` state — only `adjustTarget` (open/closed). This is the pattern from research pitfall 5.
